# Patch 19 — arthopod.sw:165

## Problem
`hatch` concatenates `getenv("HOME")` without a nil guard, producing an invalid path that is passed to `file_exists`, `file_mkdir`, and `file_write`; `generate_soul` independently repeats the same unguarded call.

## Current code
```sw
fun hatch(opts) {
    seed = getenv("USER")
    effective_seed = if (seed == nil) { "anonymous" } else { seed }
    bones = bones_from_seed(effective_seed)

    soul_path = getenv("HOME") ++ "/.swarm-code/arthopod.json"
    soul = if (file_exists(soul_path) == 'true') {
        content = file_read(soul_path)
        if (content == nil) { generate_soul(bones, soul_path, opts) }
        else {
            decoded = json_decode(content)
            if (decoded == nil) { generate_soul(bones, soul_path, opts) }
            else { decoded }
        }
    } else {
        generate_soul(bones, soul_path, opts)
    }

    map_merge(bones, soul)
}
```

## Proposed fix

```sw
fun hatch(opts) {
    seed = getenv("USER")
    effective_seed = if (seed == nil) { "anonymous" } else { seed }
    bones = bones_from_seed(effective_seed)

    home = getenv("HOME")
    effective_home = if (home == nil) { "/tmp" } else { home }
    soul_path = effective_home ++ "/.swarm-code/arthopod.json"
    soul = if (file_exists(soul_path) == 'true') {
        content = file_read(soul_path)
        if (content == nil) { generate_soul(bones, soul_path, effective_home, opts) }
        else {
            decoded = json_decode(content)
            if (decoded == nil) { generate_soul(bones, soul_path, effective_home, opts) }
            else { decoded }
        }
    } else {
        generate_soul(bones, soul_path, effective_home, opts)
    }

    map_merge(bones, soul)
}

fun generate_soul(bones, soul_path, home_dir, opts) {
    ...
    # Persist.
    file_mkdir(home_dir ++ "/.swarm-code")
    file_write(soul_path, json_encode(final_soul))
    final_soul
}
```
