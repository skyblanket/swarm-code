# Patch 04 — browser.sw:126

## Problem
`wait_for_response` recurses with the original `timeout_ms` on every event frame; steady CDP events delay match indefinitely.

## Current code
```sw
fun cdp_call_with_timeout(session, method, params, timeout_ms) {
    new_id = next_msg_id(session)
    ws = ets_get(session, 'ws')
    payload = json_encode(%{
        id: new_id,
        method: method,
        params: params
    })
    rc = wsc_send(ws, payload)
    if (rc != 'ok') { nil }
    else { wait_for_response(ws, new_id, timeout_ms) }
}

fun wait_for_response(ws, target_id, timeout_ms) {
    raw = wsc_recv(ws, timeout_ms)
    if (raw == nil) { nil }
    else {
        decoded = json_decode(raw)
        msg_id = map_get(decoded, 'id')
        if (msg_id == target_id) {
            err = map_get(decoded, 'error')
            if (err != nil) {
                # CDP reported an error; surface its message.
                em = map_get(err, 'message')
                "error: " ++ to_string(em)
            } else {
                map_get(decoded, 'result')
            }
        } else {
            # Event frame or stale — keep listening.
            wait_for_response(ws, target_id, timeout_ms)
        }
    }
}
```

## Proposed fix
```sw
fun cdp_call_with_timeout(session, method, params, timeout_ms) {
    new_id = next_msg_id(session)
    ws = ets_get(session, 'ws')
    payload = json_encode(%{
        id: new_id,
        method: method,
        params: params
    })
    rc = wsc_send(ws, payload)
    if (rc != 'ok') { nil }
    else { wait_for_response(ws, new_id, timestamp() + timeout_ms) }
}

fun wait_for_response(ws, target_id, deadline_ms) {
    remaining = deadline_ms - timestamp()
    raw = if (remaining > 0) { wsc_recv(ws, remaining) } else { nil }
    if (raw == nil) { nil }
    else {
        decoded = json_decode(raw)
        msg_id = map_get(decoded, 'id')
        if (msg_id == target_id) {
            err = map_get(decoded, 'error')
            if (err != nil) {
                # CDP reported an error; surface its message.
                em = map_get(err, 'message')
                "error: " ++ to_string(em)
            } else {
                map_get(decoded, 'result')
            }
        } else {
            # Event frame or stale — keep listening.
            wait_for_response(ws, target_id, deadline_ms)
        }
    }
}
```
